The 404 issue is because the build output directory doesn't match. Here's the instant fix:

```bash
cd ~/Desktop/projects/lush-properties-platform

# Update vercel.json to match the existing build setup
cat > vercel.json << 'EOF'
{
  "version": 2,
  "functions": {
    "api/index.js": {
      "runtime": "nodejs18.x"
    }
  },
  "builds": [
    {
      "src": "package.json",
      "use": "@vercel/static-build",
      "config": {
        "distDir": "dist/public"
      }
    }
  ],
  "routes": [
    {
      "handle": "filesystem"
    },
    {
      "src": "/api/(.*)",
      "dest": "/api/index.js"
    },
    {
      "src": "/(.*)",
      "dest": "/index.html"
    }
  ]
}
EOF

git add .
git commit -m "Fixed Vercel routing configuration"
git push origin main
```

This fixes the path mismatch between where Vite builds (`dist/public`) and where Vercel expects files (`dist`).
